#lệnh git
git -v                          check version git
git init                        tạo repo mới (không cần làm)
git clone                       truy cập repo
git status                      xem trạng thái truy cập
git add .                       thêm code lên cloud chuẩn bị commit ("." là đại diện cho thư mục hiện tại)
git add "ten_file"              tương tự git add . nhưng là chỉ định 1 file cụ thể
git commit -m                   lưu các lịch sử trong git add vào lịch sử commit (sau -m bỏ mô tả ngắn gọn về commit trong ngoặc " " )
git push                        đẩy cloud đã lưu lên github
